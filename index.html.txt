<script>
  const letter = document.getElementById('letter');
  const question = document.getElementById('question');
  const buttons = document.getElementById('buttons');
  const errorMessage = document.getElementById('errorMessage');
  const heartContainer = document.getElementById('heartContainer');
  const heartSVG = document.getElementById('heartSVG');
  const heartText = document.getElementById('heartText');
  const bgMusic = document.getElementById('bgMusic');

  // İlk mektup yavaşça görünür
  window.onload = () => {
    setTimeout(() => {
      letter.classList.add('visible');
    }, 500);

    // Mektup görünür olduktan sonra soru ve butonlar gelir
    setTimeout(() => {
      question.style.display = 'block';
      buttons.style.display = 'block';
      bgMusic.play().catch(() => {
        // Otomatik çalma engellenirse sessiz kal
      });
    }, 6000);
  };

  document.getElementById('noBtn').addEventListener('click', () => {
    buttons.style.display = 'none';
    question.style.display = 'none';
    letter.classList.remove('visible');
    errorMessage.textContent = "Buton bulunamadı. Error: 404 - Bu ihtimal hiç var olmamış gibi.";
    errorMessage.classList.add('visible');
    document.body.style.transition = 'background-color 3s';
    document.body.style.backgroundColor = '#220000';
  });

  document.getElementById('yesBtn').addEventListener('click', () => {
    buttons.style.display = 'none';
    question.style.display = 'none';
    errorMessage.classList.remove('visible');
    letter.style.display = 'none';

    // Yıldızları yavaş yavaş kaybettir
    document.getElementById('stars').style.transition = 'opacity 4s';
    document.getElementById('stars2').style.transition = 'opacity 4s';
    document.getElementById('stars3').style.transition = 'opacity 4s';
    document.getElementById('stars').style.opacity = 0;
    document.getElementById('stars2').style.opacity = 0;
    document.getElementById('stars3').style.opacity = 0;

    // Kalp ve yazıyı göster
    heartContainer.style.display = 'flex';

    // Kalp büyütme animasyonu
    setTimeout(() => {
      heartSVG.style.transform = 'scale(1)';
    }, 200);

    // Arka plan müziği devam eder
  });
</script>
